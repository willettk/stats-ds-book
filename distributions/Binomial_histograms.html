N = 100

x = np.random.normal(mu_g, sigma_g,N)

N_bins=20
mybins = np.linspace(-3,3,N_bins)
bin_to_study = int(N_bins/2)
mybins[bin_to_study], mybins[bin_to_study+1]

# probability to land in the specific bin
